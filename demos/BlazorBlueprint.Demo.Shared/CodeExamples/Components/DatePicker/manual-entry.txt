@* Culture short date or ISO (yyyy-MM-dd always works) *@
<BbDatePicker @bind-Value="_date" Editable />

@* Day-first entry: accepts dd/MM/yyyy, d/M/yyyy, or ISO *@
<BbDatePicker @bind-Value="_dateDmy" Editable
              InputFormats="@(new[] { "dd/MM/yyyy", "d/M/yyyy" })"
              DateFormat="dd/MM/yyyy"
              Placeholder="dd/mm/yyyy" />

@* Month-first entry: accepts MM/dd/yyyy, M/d/yyyy, or ISO *@
<BbDatePicker @bind-Value="_dateMdy" Editable
              InputFormats="@(new[] { "MM/dd/yyyy", "M/d/yyyy" })"
              DateFormat="MM/dd/yyyy"
              Placeholder="mm/dd/yyyy" />

@code {
    private DateTime? _date;
    private DateTime? _dateDmy;
    private DateTime? _dateMdy;
}
