@code {
    private record FileItem(string Name);
    private List<FileItem> _files = [new("report.pdf"), new("data.csv"), new("notes.txt")];

    private void HandleItemDropped(DropItemInfo<FileItem> info)
    {
        if (info.Item is null) return;
        _files.Remove(info.Item);
    }
}
