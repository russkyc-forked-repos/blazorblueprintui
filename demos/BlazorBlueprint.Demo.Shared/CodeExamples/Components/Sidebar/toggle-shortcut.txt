<!-- Ctrl/Cmd + B toggles the sidebar by default.
     Set EnableToggleShortcut="false" to disable it, for example when the
     page hosts a rich-text editor where Ctrl/Cmd + B means bold. -->
<BbSidebarProvider EnableToggleShortcut="false">
    <div class="flex h-full w-full">
        <BbSidebar>
            <!-- Sidebar content -->
        </BbSidebar>
        <BbSidebarInset>
            <!-- Main content -->
        </BbSidebarInset>
    </div>
</BbSidebarProvider>

<!-- The parameter is reactive, so it can be bound to state -->
<BbSidebarProvider EnableToggleShortcut="@(!isEditorFocused)">
    <!-- ... -->
</BbSidebarProvider>
