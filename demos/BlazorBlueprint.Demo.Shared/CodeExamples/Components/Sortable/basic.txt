@* Standalone sortable list — drag items to reorder *@
<BbDragDrop TItem="TaskItem"
            Items="_items"
            AriaLabel="Sortable task list"
            OnUpdate="@(args => SortList(_items, args))"
            Class="flex flex-col gap-2">
    <ItemTemplate Context="item">
        <BbItem Variant="ItemVariant.Outline" Class="cursor-grab active:cursor-grabbing select-none">
            <BbItemMedia>
                <LucideIcon Name="grip-vertical" Size="16" Class="text-muted-foreground" aria-hidden="true" />
            </BbItemMedia>
            <BbItemContent>
                <BbItemTitle>@item.Title</BbItemTitle>
            </BbItemContent>
        </BbItem>
    </ItemTemplate>
</BbDragDrop>

@code {
    private record TaskItem(string Title);

    private List<TaskItem> _items =
    [
        new("Design system"),
        new("Build components"),
        new("Write tests"),
        new("Publish package"),
    ];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
