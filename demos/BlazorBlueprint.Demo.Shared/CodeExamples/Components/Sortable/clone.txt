@* Pull="clone" copies items into the target; the source list is never modified.
   OnAdd on the target uses OldIndex (position in source) to insert a copy. *@
<div class="grid grid-cols-2 gap-4">
    <div class="space-y-2">
        <h3 class="text-sm font-semibold text-muted-foreground">Source (items stay)</h3>
        <BbDragDrop TItem="MyItem"
                    Items="_source"
                    Group="clone-group"
                    Pull="clone"
                    OnUpdate="@(args => SortList(_source, args))"
                    Class="flex flex-col gap-2">
            <ItemTemplate Context="item">
                <div class="flex items-center gap-2 p-3 rounded-lg border bg-card text-sm select-none cursor-grab">
                    <span aria-hidden="true">📌</span>
                    <span>@item.Title</span>
                </div>
            </ItemTemplate>
        </BbDragDrop>
    </div>
    <div class="space-y-2">
        <h3 class="text-sm font-semibold text-muted-foreground">Target (receives copies)</h3>
        <BbDragDrop TItem="MyItem"
                    Items="_target"
                    Group="clone-group"
                    OnUpdate="@(args => SortList(_target, args))"
                    OnAdd="HandleCloneAdd"
                    Class="flex flex-col gap-2 min-h-[60px] rounded-lg border-2 border-dashed border-muted-foreground/30 p-2">
            <ItemTemplate Context="item">
                <div class="flex items-center gap-2 p-3 rounded-lg border bg-card text-sm select-none cursor-grab">
                    <span aria-hidden="true">📎</span>
                    <span>@item.Title</span>
                </div>
            </ItemTemplate>
        </BbDragDrop>
    </div>
</div>

@code {
    private record MyItem(string Title);

    private List<MyItem> _source = [new("Template A"), new("Template B"), new("Template C")];
    private List<MyItem> _target = [];

    // OldIndex is the position in _source (unchanged in clone mode).
    private void HandleCloneAdd((int OldIndex, int NewIndex) args)
    {
        if (args.OldIndex < 0 || args.OldIndex >= _source.Count) { return; }
        _target.Insert(Math.Clamp(args.NewIndex, 0, _target.Count), _source[args.OldIndex]);
    }

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
