@* Clone mode — Pull="clone" copies items to the target; OnRemove never fires on the source.
   OnAdd receives the item directly — insert it at args.NewIndex. *@
<div class="grid grid-cols-2 gap-4">
    <div class="space-y-2">
        <p class="text-sm font-medium text-muted-foreground">Source — items stay</p>
        <BbDragDrop TItem="MyItem"
                    Items="_source"
                    Group="clone-demo"
                    Pull="clone"
                    AriaLabel="Clone source list"
                    OnUpdate="@(args => SortList(_source, args))"
                    Class="flex flex-col gap-2">
            <ItemTemplate Context="item">
                <BbItem Variant="ItemVariant.Outline" Class="cursor-grab active:cursor-grabbing select-none">
                    <BbItemMedia>
                        <LucideIcon Name="grip-vertical" Size="16" Class="text-muted-foreground" aria-hidden="true" />
                    </BbItemMedia>
                    <BbItemContent>
                        <BbItemTitle>@item.Title</BbItemTitle>
                    </BbItemContent>
                </BbItem>
            </ItemTemplate>
        </BbDragDrop>
    </div>
    <div class="space-y-2">
        <p class="text-sm font-medium text-muted-foreground">Target — receives copies</p>
        <BbDragDrop TItem="MyItem"
                    Items="_target"
                    Group="clone-demo"
                    AriaLabel="Clone target list"
                    OnUpdate="@(args => SortList(_target, args))"
                    OnAdd="@(args => _target.Insert(Math.Clamp(args.NewIndex, 0, _target.Count), args.Item))"
                    Class="flex flex-col gap-2 min-h-[60px] rounded-lg border-2 border-dashed
                           border-muted-foreground/30 p-2">
            <ItemTemplate Context="item">
                <BbItem Variant="ItemVariant.Outline" Class="cursor-grab active:cursor-grabbing select-none">
                    <BbItemMedia>
                        <LucideIcon Name="grip-vertical" Size="16" Class="text-muted-foreground" aria-hidden="true" />
                    </BbItemMedia>
                    <BbItemContent>
                        <BbItemTitle>@item.Title</BbItemTitle>
                    </BbItemContent>
                </BbItem>
            </ItemTemplate>
        </BbDragDrop>
    </div>
</div>

@code {
    private record MyItem(string Title);

    private List<MyItem> _source = [new("Template A"), new("Template B"), new("Template C")];
    private List<MyItem> _target = [];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
