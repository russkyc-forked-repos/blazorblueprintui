@* ItemDisabled — predicate returns true for items that cannot be dragged.
   Disabled items render with draggable="false" and no drag ARIA attributes. *@
<BbDragDrop TItem="PrefItem"
            Items="_items"
            ItemDisabled="@(item => item.Locked)"
            AriaLabel="Sortable preferences list"
            OnUpdate="@(args => SortList(_items, args))"
            Class="flex flex-col gap-2">
    <ItemTemplate Context="item">
        <BbItem Variant="ItemVariant.Outline"
                Class="@(item.Locked ? "opacity-60 cursor-not-allowed select-none" : "cursor-grab active:cursor-grabbing select-none")">
            <BbItemMedia>
                @if (item.Locked)
                {
                    <LucideIcon Name="lock" Size="16" Class="text-muted-foreground" aria-hidden="true" />
                }
                else
                {
                    <LucideIcon Name="grip-vertical" Size="16" Class="text-muted-foreground" aria-hidden="true" />
                }
            </BbItemMedia>
            <BbItemContent>
                <BbItemTitle>@item.Title</BbItemTitle>
            </BbItemContent>
            @if (item.Locked)
            {
                <BbItemActions>
                    <BbBadge Variant="BadgeVariant.Secondary">
                        <LucideIcon Name="lock" Size="10" Class="mr-1" aria-hidden="true" />
                        Locked
                    </BbBadge>
                </BbItemActions>
            }
        </BbItem>
    </ItemTemplate>
</BbDragDrop>

@code {
    private record PrefItem(string Title, bool Locked);

    private List<PrefItem> _items =
    [
        new("Required: Terms of Service", Locked: true),
        new("Optional: Marketing emails",  Locked: false),
        new("Required: Privacy policy",    Locked: true),
        new("Optional: Newsletter",        Locked: false),
    ];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
