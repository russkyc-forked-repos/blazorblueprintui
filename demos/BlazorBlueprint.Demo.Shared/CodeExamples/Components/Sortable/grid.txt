@* Grid sortable — pass any CSS grid class via the Class parameter *@
<BbDragDrop TItem="GridItem"
            Items="_photos"
            OnUpdate="@(args => SortList(_photos, args))"
            Class="grid grid-cols-4 gap-3">
    <ItemTemplate Context="photo">
        <div class="flex items-center justify-center rounded-lg border bg-card shadow-sm select-none cursor-grab active:cursor-grabbing aspect-square text-3xl hover:shadow-md transition-shadow">
            @photo.Emoji
        </div>
    </ItemTemplate>
</BbDragDrop>

@code {
    private record GridItem(string Emoji);

    private List<GridItem> _photos =
    [
        new("🌅"), new("🏔️"), new("🌊"), new("🌲"),
        new("🌺"), new("🦋"), new("🌙"), new("⭐"),
        new("🌈"), new("🍀"), new("🔥"), new("❄️"),
    ];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
