@* Set Handle to a CSS class to restrict dragging to a specific element inside each item *@
<BbDragDrop TItem="MyItem"
            Items="_items"
            Handle="drag-handle"
            OnUpdate="@(args => SortList(_items, args))"
            Class="flex flex-col gap-2">
    <ItemTemplate Context="item">
        <div class="flex items-center gap-3 p-3 rounded-lg border bg-card select-none">
            @* Only this element initiates drag *@
            <span class="drag-handle cursor-grab text-muted-foreground hover:text-foreground" aria-hidden="true">⠿</span>
            <span class="text-sm font-medium">@item.Title</span>
        </div>
    </ItemTemplate>
</BbDragDrop>

@code {
    private record MyItem(string Title);

    private List<MyItem> _items = [new("Item A"), new("Item B"), new("Item C")];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
