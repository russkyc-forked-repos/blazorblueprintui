@* Handle — only elements with the Handle CSS class can initiate a drag.
   Wrap the handle in a <button> with a descriptive aria-label for accessibility. *@
<BbDragDrop TItem="TaskItem"
            Items="_items"
            Handle="bb-drag-handle"
            AriaLabel="Sortable list with drag handles"
            OnUpdate="@(args => SortList(_items, args))"
            Class="flex flex-col gap-2">
    <ItemTemplate Context="item">
        <BbItem Variant="ItemVariant.Outline" Class="select-none">
            <BbItemMedia>
                <button class="bb-drag-handle cursor-grab text-muted-foreground hover:text-foreground
                               transition-colors focus-visible:outline-none focus-visible:ring-2
                               focus-visible:ring-ring rounded"
                        aria-label="Drag to reorder @item.Title"
                        type="button">
                    <LucideIcon Name="grip-vertical" Size="16" aria-hidden="true" />
                </button>
            </BbItemMedia>
            <BbItemContent>
                <BbItemTitle>@item.Title</BbItemTitle>
            </BbItemContent>
        </BbItem>
    </ItemTemplate>
</BbDragDrop>

@code {
    private record TaskItem(string Title);

    private List<TaskItem> _items =
    [
        new("Design system"), new("Build components"), new("Write tests"),
    ];

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
