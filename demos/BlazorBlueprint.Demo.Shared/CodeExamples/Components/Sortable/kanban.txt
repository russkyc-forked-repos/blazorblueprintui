@* Kanban board: drag cards between columns and reorder within each *@
<BbDropContainer T="KanbanCard"
                 ItemDropped="HandleItemDropped">
    <div class="grid grid-cols-3 gap-4">
        @foreach (var column in _columns)
        {
            <div class="flex flex-col gap-2 rounded-lg border bg-muted/30 p-3">
                <h3 class="text-sm font-semibold text-muted-foreground px-1">@column.Name</h3>
                <BbSortable T="KanbanCard"
                            Items="column.Cards"
                            ZoneIdentifier="@column.Id"
                            AllowReorder="true"
                            AriaLabel="@column.Name"
                            OnItemDropped="HandleItemDropped">
                    <ItemTemplate Context="card">
                        <BbCard Class="cursor-grab active:cursor-grabbing">
                            <BbCardContent Class="p-3">
                                <p class="text-sm font-medium">@card.Title</p>
                                <p class="text-xs text-muted-foreground mt-1">@card.Description</p>
                            </BbCardContent>
                        </BbCard>
                    </ItemTemplate>
                </BbSortable>
            </div>
        }
    </div>
</BbDropContainer>

@code {
    private record KanbanCard(string Id, string Title, string Description, string Column);

    private class KanbanColumn(string id, string name)
    {
        public string Id { get; } = id;
        public string Name { get; } = name;
        public List<KanbanCard> Cards { get; set; } = [];
    }

    private List<KanbanColumn> _columns =
    [
        new("todo", "To Do") { Cards = [new("1","Design mockups","Create UI wireframes","todo"), new("2","Write specs","Document requirements","todo")] },
        new("doing", "In Progress") { Cards = [new("3","Build API","REST endpoints","doing")] },
        new("done", "Done") { Cards = [new("4","Project setup","Scaffold repository","done")] },
    ];

    private void HandleItemDropped(DropItemInfo<KanbanCard> info)
    {
        if (info.Item is null) return;

        // Remove from source column
        var src = _columns.First(c => c.Id == info.SourceZone);
        src.Cards.Remove(info.Item);

        // Insert into target column at the correct index
        var dst = _columns.First(c => c.Id == info.TargetZone);
        var updated = info.Item with { Column = info.TargetZone };
        var idx = Math.Clamp(info.TargetIndex, 0, dst.Cards.Count);
        dst.Cards.Insert(idx, updated);
    }
}
