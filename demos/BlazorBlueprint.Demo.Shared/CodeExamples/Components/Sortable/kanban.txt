@* Kanban board — multiple lists share the same Group name.
   OnAdd receives the dropped item directly; no tracking variable needed.
   OnRemove fires on the source after a successful drop. *@
<div class="grid grid-cols-3 gap-4">
    @foreach (var col in _columns)
    {
        var column = col;
        <BbCard Class="flex flex-col min-h-[220px]">
            <BbCardHeader Class="pb-2">
                <div class="flex items-center justify-between">
                    <BbCardTitle Class="text-sm font-semibold">@column.Name</BbCardTitle>
                    <BbBadge Variant="BadgeVariant.Secondary">@column.Cards.Count</BbBadge>
                </div>
            </BbCardHeader>
            <BbCardContent Class="flex-1 pt-0 px-3 pb-3">
                <BbDragDrop TItem="KanbanCard"
                            Items="column.Cards"
                            Group="kanban"
                            AriaLabel="@($"{column.Name} column")"
                            OnUpdate="@(args => SortList(column.Cards, args))"
                            OnRemove="@(args => column.Cards.RemoveAt(args.OldIndex))"
                            OnAdd="@(args => column.Cards.Insert(Math.Clamp(args.NewIndex, 0, column.Cards.Count), args.Item))"
                            Class="flex flex-col gap-2 min-h-[3rem]">
                    <ItemTemplate Context="card">
                        <BbCard Class="cursor-grab active:cursor-grabbing shadow-sm hover:shadow-md transition-shadow">
                            <BbCardContent Class="p-3">
                                <p class="text-sm font-medium">@card.Title</p>
                                <p class="text-xs text-muted-foreground mt-1">@card.Description</p>
                            </BbCardContent>
                        </BbCard>
                    </ItemTemplate>
                </BbDragDrop>
            </BbCardContent>
        </BbCard>
    }
</div>

@code {
    private record KanbanCard(string Id, string Title, string Description);

    private sealed class KanbanColumn(string id, string name)
    {
        public string Id   { get; } = id;
        public string Name { get; } = name;
        public List<KanbanCard> Cards { get; } = [];
    }

    private List<KanbanColumn> _columns =
    [
        new("todo", "To Do"), new("doing", "In Progress"), new("done", "Done"),
    ];

    protected override void OnInitialized()
    {
        _columns[0].Cards.AddRange(new[]
        {
            new KanbanCard("1", "Design mockups",   "UI wireframes"),
            new KanbanCard("2", "Write specs",      "API surface docs"),
        });
        _columns[1].Cards.Add(new KanbanCard("3", "Build API", "REST endpoints"));
        _columns[2].Cards.Add(new KanbanCard("4", "Project setup", "Scaffold repo"));
    }

    private static void SortList<T>(IList<T> list, (int OldIndex, int NewIndex) args)
    {
        var (oldIndex, newIndex) = args;
        if (oldIndex < 0 || oldIndex >= list.Count) { return; }
        var item = list[oldIndex];
        list.RemoveAt(oldIndex);
        if (newIndex < list.Count) { list.Insert(newIndex, item); }
        else { list.Add(item); }
    }
}
